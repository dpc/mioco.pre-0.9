Coroutine configuration. Provides detailed control over the properties and behavior of new Coroutines.

let coro = Builder::new().name(format!("Coroutine #{}", 1))
                         .stack_size(4096)
                         .spawn(|| println!("Hello world!!"));

coro.resume().unwrap();
